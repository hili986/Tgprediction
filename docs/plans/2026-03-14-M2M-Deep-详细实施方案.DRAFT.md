### 2.1 `src/features/physical_proxy.py`

```python
"""
Physical Proxy Features (PPF) for Tg Prediction
物理代理特征 — 从单体推断材料行为

10 features bridging monomer structure to material behavior:
    PPF_1  M_per_f           — mass per flexible bond (Gibbs-DiMarzio)
    PPF_2  CED_estimate      — cohesive energy density (Van Krevelen GC)
    PPF_3  Vf_estimate       — free volume fraction estimate
    PPF_4  packing_efficiency — V_vdW / V_m ratio
    PPF_5  side_chain_ratio  — side chain atoms / backbone atoms
    PPF_6  steric_volume     — side group Van der Waals volume
    PPF_7  symmetry_index    — structural symmetry score [0, 1]
    PPF_8  CED_hbond_frac    — H-bond contribution to CED
    PPF_9  backbone_rigidity — fraction of rigid backbone bonds
    PPF_10 flexible_bond_density — flexible bonds / total bonds

Public API:
    compute_ppf(smiles) -> dict
    ppf_feature_names() -> list[str]
    ppf_vector(smiles) -> list[float]
"""

from typing import Dict, List, Set, Tuple
import math
from rdkit import Chem
from rdkit.Chem import Descriptors, AllChem, rdMolDescriptors

FEATURE_NAMES = [
    "M_per_f", "CED_estimate", "Vf_estimate", "packing_efficiency",
    "side_chain_ratio", "steric_volume", "symmetry_index",
    "CED_hbond_frac", "backbone_rigidity", "flexible_bond_density",
]

# ── 柔性键识别规则 ──────────────────────────────────
# 遍历 mol.GetBonds()，对每个非环单键:
#   C-C 非环单键: f=1.0
#   C-O/O-C 非环单键: f=1.0
#   Si-O: f=1.5 (有机硅超柔)
#   酰胺旁 C-N (邻位有 C=O): f=0 (共轭限制旋转)
#   芳环间单键 (两端 GetIsAromatic): f=0.5
#   环内键: f=0
#   双键/三键: f=0

def _count_flexible_bonds(mol: Chem.Mol) -> float:
    """Count flexible bonds with type-specific weights."""
    f_total = 0.0
    for bond in mol.GetBonds():
        if bond.IsInRing():
            continue
        if bond.GetBondType() != Chem.BondType.SINGLE:
            continue
        a1 = bond.GetBeginAtom()
        a2 = bond.GetEndAtom()
        s1, s2 = a1.GetSymbol(), a2.GetSymbol()

        # 检测酰胺键旁: C-N 且 N 邻位有 C=O
        if _is_amide_adjacent(a1, a2, bond):
            continue  # f=0

        # 芳环间单键
        if a1.GetIsAromatic() and a2.GetIsAromatic():
            f_total += 0.5
            continue

        # Si-O
        if (s1 == "Si" and s2 == "O") or (s1 == "O" and s2 == "Si"):
            f_total += 1.5
            continue

        # C-C, C-O 非环单键
        if {s1, s2} <= {"C", "O", "N", "S"}:
            f_total += 1.0

    return max(f_total, 0.5)  # 至少 0.5 防除零


def _is_amide_adjacent(a1, a2, bond) -> bool:
    """Check if this C-N bond is adjacent to C=O (amide-like)."""
    for atom in [a1, a2]:
        if atom.GetSymbol() == "N":
            for nbr in atom.GetNeighbors():
                if nbr.GetIdx() == bond.GetOtherAtomIdx(atom.GetIdx()):
                    continue
                if nbr.GetSymbol() == "C":
                    for b in nbr.GetBonds():
                        other = b.GetOtherAtom(nbr)
                        if (other.GetSymbol() == "O" and
                            b.GetBondType() == Chem.BondType.DOUBLE):
                            return True
    return False


def _find_backbone_atoms(mol: Chem.Mol, smiles: str) -> Set[int]:
    """Find backbone atoms: shortest path between * attachment points."""
    # 找到 dummy atoms 或 [H] 替换后的端点
    # 对于 *CC(C)* → CC(C) 的情况，端点是第一个和最后一个碳
    # 策略：如果原始 smiles 有 *，找到 * 邻居作为端点
    dummy_neighbors = []
    if "*" in smiles:
        raw_mol = Chem.MolFromSmiles(smiles)
        if raw_mol is not None:
            for atom in raw_mol.GetAtoms():
                if atom.GetAtomicNum() == 0:  # dummy atom *
                    for nbr in atom.GetNeighbors():
                        dummy_neighbors.append(nbr.GetIdx())

    if len(dummy_neighbors) >= 2:
        # 在 clean mol 中找对应原子（index 可能不同）
        # Fallback: 用度最高的两个非氢原子作为端点
        path = Chem.rdmolops.GetShortestPath(mol, 0, mol.GetNumAtoms() - 1)
        return set(path)

    # Fallback: Murcko scaffold 的原子
    from rdkit.Chem.Scaffolds import MurckoScaffold
    try:
        scaffold = MurckoScaffold.GetScaffoldForMol(mol)
        scaffold_smiles = Chem.MolToSmiles(scaffold)
        scaffold_mol = Chem.MolFromSmiles(scaffold_smiles)
        if scaffold_mol:
            match = mol.GetSubstructMatch(scaffold_mol)
            return set(match)
    except Exception:
        pass

    return set(range(mol.GetNumHeavyAtoms()))


def _compute_symmetry_index(mol: Chem.Mol, smiles: str) -> float:
    """Compute structural symmetry score [0, 1].

    Method: For each substituted backbone carbon, compare substituents
    on both sides. More similar substituents → higher symmetry.

    Examples:
      PE (*CC*):            1.0 (CH2-CH2, fully symmetric)
      PP (*CC(C)*):         0.0 (CH2-CH(CH3), asymmetric)
      PIB (*CC(C)(C)*):     0.5 (CH2 sym, C(CH3)2 sym)
      PVDC (*CC(Cl)(Cl)*):  0.5
    """
    backbone = _find_backbone_atoms(mol, smiles)
    n_substituted = 0
    n_symmetric = 0

    for idx in backbone:
        atom = mol.GetAtomWithIdx(idx)
        if atom.GetSymbol() != "C":
            continue

        # 收集非主链、非氢取代基
        subs = []
        for nbr in atom.GetNeighbors():
            if nbr.GetIdx() not in backbone and nbr.GetAtomicNum() > 1:
                # 获取取代基的 SMILES 子结构
                subs.append(nbr.GetAtomicNum())

        if len(subs) == 0:
            continue  # 无取代基

        n_substituted += 1

        if len(subs) >= 2:
            # 两个取代基相同 → 对称
            if subs[0] == subs[1]:
                n_symmetric += 1
        # 单取代 → 不对称（n_symmetric 不增加）

    if n_substituted == 0:
        return 1.0  # 无取代 = 完全对称（如 PE）

    return n_symmetric / n_substituted


def compute_ppf(smiles: str) -> Dict[str, float]:
    """Compute all 10 PPF features."""
    clean = smiles.replace("*", "[H]")
    mol = Chem.MolFromSmiles(clean)
    if mol is None:
        return {name: float("nan") for name in FEATURE_NAMES}

    mw = Descriptors.MolWt(mol)
    heavy = Descriptors.HeavyAtomCount(mol) or 1

    # PPF_1: M/f
    f = _count_flexible_bonds(mol)
    m_per_f = mw / f

    # PPF_2: CED (复用 solubility_param 的 GC 表)
    from src.features.solubility_param import compute_solubility_param
    sol = compute_solubility_param(mol)
    ced = sol ** 2  # CED = delta^2

    # PPF_3: Vf estimate = 1 - 1.3 * V_vdW / V_m
    v_vdw = Descriptors.LabuteASA(mol)  # 近似 Van der Waals 表面积
    # 用 MolWt / 1.0 作为 V_m 的粗估（密度 ≈ 1 g/cm³）
    v_m = mw  # 粗估
    vf = 1.0 - 1.3 * v_vdw / max(v_m, 1.0)
    vf = max(0.0, min(1.0, vf))

    # PPF_4: packing efficiency
    packing = v_vdw / max(v_m, 1.0)

    # PPF_5: side chain ratio
    backbone = _find_backbone_atoms(mol, smiles)
    n_backbone = len(backbone) or 1
    n_sidechain = heavy - n_backbone
    sc_ratio = max(n_sidechain, 0) / n_backbone

    # PPF_6: steric volume (侧基 heavy atom count)
    steric = n_sidechain * 15.0  # 粗估每个重原子 ~15 Å³

    # PPF_7: symmetry index
    sym = _compute_symmetry_index(mol, smiles)

    # PPF_8: CED H-bond fraction
    from src.features.solubility_param import GC_GROUPS, _COMPILED_PATTERNS
    hbond_ecoh = 0.0
    total_ecoh = 0.0
    hbond_groups = {"[OX2H]", "[NX3H2]", "[NX3H1]", "[NX3]([CX3]=O)"}
    for pattern, name, ecoh, vw in _COMPILED_PATTERNS:
        matches = mol.GetSubstructMatches(pattern)
        count = len(matches)
        total_ecoh += ecoh * count
        # 检查是否是氢键基团
        for smarts_key, (gname, gecoh, gvw) in GC_GROUPS.items():
            if gecoh == ecoh and smarts_key in hbond_groups:
                hbond_ecoh += ecoh * count
                break
    ced_hbond_frac = hbond_ecoh / max(total_ecoh, 1.0)

    # PPF_9: backbone rigidity (芳环/双键在主链中的比例)
    rigid_bonds = 0
    total_backbone_bonds = 0
    for bond in mol.GetBonds():
        a1, a2 = bond.GetBeginAtomIdx(), bond.GetEndAtomIdx()
        if a1 in backbone and a2 in backbone:
            total_backbone_bonds += 1
            if bond.GetIsAromatic() or bond.GetBondType() == Chem.BondType.DOUBLE:
                rigid_bonds += 1
    br = rigid_bonds / max(total_backbone_bonds, 1)

    # PPF_10: flexible bond density
    total_bonds = mol.GetNumBonds() or 1
    fbd = f / total_bonds

    return {
        "M_per_f": m_per_f,
        "CED_estimate": ced,
        "Vf_estimate": vf,
        "packing_efficiency": packing,
        "side_chain_ratio": sc_ratio,
        "steric_volume": steric,
        "symmetry_index": sym,
        "CED_hbond_frac": ced_hbond_frac,
        "backbone_rigidity": br,
        "flexible_bond_density": fbd,
    }


def ppf_feature_names() -> List[str]:
    return list(FEATURE_NAMES)


def ppf_vector(smiles: str) -> List[float]:
    features = compute_ppf(smiles)
    return [features[name] for name in FEATURE_NAMES]
```

### 2.2 `src/features/virtual_polymerization.py`

```python
"""
Virtual Polymerization Descriptors (VPD) for Tg Prediction
虚拟聚合描述符 — 在寡聚物上计算描述符捕获聚合效应

12 features:
    VPD_1~4   三聚体归一化描述符 (RotBonds, TPSA, LogP, Kappa2 per RU)
    VPD_5~8   聚合效应 Δ = (dimer/2) - monomer
    VPD_9~12  连接点特征 (bond_type, steric, conjugation, symmetry)

Public API:
    compute_vpd(smiles) -> dict
    vpd_feature_names() -> list[str]
    vpd_vector(smiles) -> list[float]
    build_oligomer(repeat_smiles, n) -> str | None
"""

from typing import Dict, List, Optional
from rdkit import Chem
from rdkit.Chem import AllChem, Descriptors, RWMol


FEATURE_NAMES = [
    "tri_RotBonds_per_RU", "tri_TPSA_per_RU",
    "tri_LogP_per_RU", "tri_Kappa2_per_RU",
    "delta_RotBonds", "delta_TPSA", "delta_FrCSP3", "delta_Kappa2",
    "junc_bond_type", "junc_steric", "junc_conjugation", "junc_symmetry",
]

# 用于计算的 4 个核心描述符
_CORE_DESCS = {
    "RotBonds": Descriptors.NumRotatableBonds,
    "TPSA": Descriptors.TPSA,
    "LogP": Descriptors.MolLogP,
    "Kappa2": Descriptors.Kappa2,
}


def build_oligomer(repeat_smiles: str, n: int = 3) -> Optional[str]:
    """Assemble n copies of repeat unit into an oligomer.

    Algorithm:
    1. Parse SMILES, find exactly 2 dummy atoms (*)
    2. Record attachment point neighbors and bond types
    3. Clone n copies of the monomer
    4. Connect adjacent copies: right-* neighbor of copy[i] ↔ left-* neighbor of copy[i+1]
    5. Remove all dummy atoms, cap with H
    6. SanitizeMol, return canonical SMILES

    Returns None if assembly fails.
    """
    mol = Chem.MolFromSmiles(repeat_smiles)
    if mol is None:
        return None

    # 找 dummy atoms (atomic number = 0)
    dummies = [a.GetIdx() for a in mol.GetAtoms() if a.GetAtomicNum() == 0]
    if len(dummies) != 2:
        # 没有 * 标记，尝试用首尾原子
        return _fallback_oligomer(repeat_smiles, n)

    # 记录每个 dummy 的邻居和键类型
    d_left, d_right = dummies[0], dummies[1]
    left_info = _get_attachment_info(mol, d_left)
    right_info = _get_attachment_info(mol, d_right)

    if left_info is None or right_info is None:
        return None

    try:
        # 使用 CombineMols 合并 n 个拷贝
        combined = Chem.RWMol(mol)
        offsets = [0]
        for i in range(1, n):
            combined = Chem.RWMol(Chem.CombineMols(combined, mol))
            offsets.append(offsets[-1] + mol.GetNumAtoms())

        # 连接相邻拷贝
        for i in range(n - 1):
            # copy[i] 的右 dummy 邻居 ↔ copy[i+1] 的左 dummy 邻居
            right_nbr = offsets[i] + right_info["neighbor_idx"]
            left_nbr = offsets[i + 1] + left_info["neighbor_idx"]
            bond_type = right_info["bond_type"]
            combined.AddBond(right_nbr, left_nbr, bond_type)

        # 移除所有 dummy atoms (从大 index 开始)
        all_dummies = sorted(
            [a.GetIdx() for a in combined.GetAtoms() if a.GetAtomicNum() == 0],
            reverse=True
        )
        for d_idx in all_dummies:
            combined.RemoveAtom(d_idx)

        # Sanitize
        Chem.SanitizeMol(combined)
        return Chem.MolToSmiles(combined)

    except Exception:
        return _fallback_oligomer(repeat_smiles, n)


def _get_attachment_info(mol, dummy_idx):
    """Get info about the attachment point of a dummy atom."""
    atom = mol.GetAtomWithIdx(dummy_idx)
    neighbors = atom.GetNeighbors()
    if len(neighbors) != 1:
        return None
    nbr = neighbors[0]
    bond = mol.GetBondBetweenAtoms(dummy_idx, nbr.GetIdx())
    return {
        "neighbor_idx": nbr.GetIdx(),
        "bond_type": bond.GetBondType(),
        "neighbor_symbol": nbr.GetSymbol(),
    }


def _fallback_oligomer(smiles: str, n: int) -> Optional[str]:
    """Fallback: simple string concatenation for * -free SMILES."""
    clean = smiles.replace("*", "")
    if not clean:
        return None
    try:
        repeated = clean * n
        mol = Chem.MolFromSmiles(repeated)
        if mol is not None:
            return Chem.MolToSmiles(mol)
    except Exception:
        pass
    return None


def _compute_core_descs(smiles_or_mol) -> Dict[str, float]:
    """Compute 4 core descriptors on a molecule."""
    if isinstance(smiles_or_mol, str):
        mol = Chem.MolFromSmiles(smiles_or_mol)
    else:
        mol = smiles_or_mol

    if mol is None:
        return {k: float("nan") for k in _CORE_DESCS}

    result = {}
    for name, func in _CORE_DESCS.items():
        try:
            result[name] = float(func(mol))
        except Exception:
            result[name] = float("nan")
    return result


def _compute_junction_features(smiles: str) -> Dict[str, float]:
    """Extract attachment point (junction) features from repeat SMILES."""
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return {"junc_bond_type": 0, "junc_steric": 0,
                "junc_conjugation": 0, "junc_symmetry": 0}

    dummies = [a for a in mol.GetAtoms() if a.GetAtomicNum() == 0]

    if len(dummies) < 2:
        return {"junc_bond_type": 0, "junc_steric": 0,
                "junc_conjugation": 0, "junc_symmetry": 0}

    d = dummies[0]
    nbrs = d.GetNeighbors()
    if not nbrs:
        return {"junc_bond_type": 0, "junc_steric": 0,
                "junc_conjugation": 0, "junc_symmetry": 0}

    nbr = nbrs[0]

    # bond_type: C=0, O=1, N=2, aromatic=3, other=4
    sym = nbr.GetSymbol()
    bt_map = {"C": 0, "O": 1, "N": 2, "S": 3}
    bond_type = bt_map.get(sym, 4)
    if nbr.GetIsAromatic():
        bond_type = 3

    # steric: non-H neighbor count of attachment atom
    steric = sum(1 for n in nbr.GetNeighbors()
                 if n.GetAtomicNum() > 1 and n.GetAtomicNum() != 0)

    # conjugation: is attachment atom in aromatic/conjugated system
    conjugation = 1.0 if nbr.GetIsAromatic() else 0.0

    # symmetry: are the two attachment points equivalent
    d2 = dummies[1]
    nbrs2 = d2.GetNeighbors()
    sym_score = 0.0
    if nbrs2:
        nbr2 = nbrs2[0]
        if nbr.GetSymbol() == nbr2.GetSymbol():
            sym_score = 0.5
            if nbr.GetDegree() == nbr2.GetDegree():
                sym_score = 1.0

    return {
        "junc_bond_type": float(bond_type),
        "junc_steric": float(steric),
        "junc_conjugation": conjugation,
        "junc_symmetry": sym_score,
    }


def compute_vpd(smiles: str) -> Dict[str, float]:
    """Compute all 12 VPD features."""
    # 单体描述符 (封端后)
    mono_smiles = smiles.replace("*", "[H]")
    mono_descs = _compute_core_descs(mono_smiles)

    # 二聚体
    dimer_smiles = build_oligomer(smiles, n=2)
    dimer_descs = _compute_core_descs(dimer_smiles) if dimer_smiles else mono_descs

    # 三聚体
    trimer_smiles = build_oligomer(smiles, n=3)
    trimer_descs = _compute_core_descs(trimer_smiles) if trimer_smiles else mono_descs

    # 连接点特征
    junc = _compute_junction_features(smiles)

    result = {}

    # VPD_1~4: 三聚体归一化
    for key in ["RotBonds", "TPSA", "LogP", "Kappa2"]:
        tri_name = f"tri_{key}_per_RU"
        result[tri_name] = trimer_descs[key] / 3.0

    # VPD_5~8: 聚合效应 Δ
    delta_keys = ["RotBonds", "TPSA", "FrCSP3", "Kappa2"]
    # FrCSP3 需要单独算
    mono_frcsp3 = Descriptors.FractionCSP3(Chem.MolFromSmiles(mono_smiles) or Chem.MolFromSmiles("C"))
    if dimer_smiles:
        dimer_mol = Chem.MolFromSmiles(dimer_smiles)
        dimer_frcsp3 = Descriptors.FractionCSP3(dimer_mol) if dimer_mol else mono_frcsp3
    else:
        dimer_frcsp3 = mono_frcsp3

    result["delta_RotBonds"] = (dimer_descs["RotBonds"] / 2.0) - mono_descs["RotBonds"]
    result["delta_TPSA"] = (dimer_descs["TPSA"] / 2.0) - mono_descs["TPSA"]
    result["delta_FrCSP3"] = dimer_frcsp3 - mono_frcsp3
    result["delta_Kappa2"] = (dimer_descs["Kappa2"] / 2.0) - mono_descs["Kappa2"]

    # VPD_9~12: 连接点
    result.update(junc)

    return result


def vpd_feature_names() -> List[str]:
    return list(FEATURE_NAMES)


def vpd_vector(smiles: str) -> List[float]:
    features = compute_vpd(smiles)
    return [features[name] for name in FEATURE_NAMES]
```

```python
"""
Physics-Constrained GBR for Tg Prediction
物理约束梯度提升 — 单调性约束 + 值域约束

First application of monotone constraints to Tg prediction.
2025 PINN review confirms no prior work.
"""

from typing import Dict, List, Optional, Tuple
import numpy as np
from sklearn.ensemble import GradientBoostingRegressor

# 从物理定律推导的单调性约束
PHYSICS_MONOTONE: Dict[str, int] = {
    "PPF_M_per_f": 1,              # M/f ↑ → Tg ↑ (Gibbs-DiMarzio)
    "PPF_CED_estimate": 1,         # CED ↑ → Tg ↑
    "PPF_backbone_rigidity": 1,    # 刚性 ↑ → Tg ↑
    "L0_FlexibilityIndex": -1,     # 柔性 ↑ → Tg ↓
    "L1_RingCount": 1,             # 环多 → Tg ↑
    "L1_NumAromaticRings": 1,      # 芳环 → Tg ↑
    "PPF_Vf_estimate": -1,         # 自由体积 ↑ → Tg ↓
    "PPF_symmetry_index": -1,      # 对称 ↑ → Tg ↓
    "PPF_flexible_bond_density": -1, # 柔性键密度 ↑ → Tg ↓
}


def build_constrained_gbr(
    feature_names: List[str],
    custom_constraints: Optional[Dict[str, int]] = None,
    **gbr_kwargs,
) -> GradientBoostingRegressor:
    """Build GBR with physics-derived monotone constraints."""
    constraints = custom_constraints or PHYSICS_MONOTONE

    mono = []
    for name in feature_names:
        mono.append(constraints.get(name, 0))

    default_params = {
        "n_estimators": 300,
        "max_depth": 5,
        "learning_rate": 0.05,
        "subsample": 0.8,
        "min_samples_leaf": 5,
    }
    default_params.update(gbr_kwargs)

    return GradientBoostingRegressor(
        monotone_constraints=mono,
        **default_params,
    )


def clip_predictions(
    y_pred: np.ndarray,
    min_tg: float = 100.0,
    max_tg: float = 700.0,
) -> np.ndarray:
    """Apply physical range constraint."""
    return np.clip(y_pred, min_tg, max_tg)
```

### 3.2 `src/ml/hierarchical_model.py`

```python
"""
Hierarchical Residual Learning (HRL) for Tg Prediction
层级残差学习 — 4 层物理分解

Tg = L0(backbone) + L1(steric) + L2(polar) + L3(residual)

Each layer only learns the residual from previous layers.
Unlike Stacking (which failed 3x on 304 samples), HRL's
hierarchy comes from physics, not statistics.
"""

from typing import Any, Dict, List, Optional
import numpy as np
from sklearn.linear_model import LinearRegression
from sklearn.ensemble import GradientBoostingRegressor
from sklearn.model_selection import RepeatedKFold

# 每个物理层级使用的特征（从 feature_names 中匹配）
HRL_FEATURE_GROUPS = {
    "L0_backbone": ["PPF_M_per_f"],
    "L1_steric": ["PPF_steric_volume", "PPF_symmetry_index", "PPF_side_chain_ratio"],
    "L2_polar": ["PPF_CED_estimate", "PPF_CED_hbond_frac", "L0_HBondDensity"],
}


class HierarchicalTgPredictor:
    def __init__(
        self,
        l1_n_estimators: int = 50,
        l2_n_estimators: int = 50,
        l3_n_estimators: int = 200,
        l3_max_depth: int = 5,
    ):
        self.layer0 = LinearRegression()
        self.layer1 = GradientBoostingRegressor(
            n_estimators=l1_n_estimators, max_depth=3,
            monotone_constraints=[1, -1, 0],  # steric↑Tg, sym↓Tg
        )
        self.layer2 = GradientBoostingRegressor(
            n_estimators=l2_n_estimators, max_depth=3,
            monotone_constraints=[1, 1, 1],   # CED↑Tg, HBond↑Tg
        )
        self.layer3 = GradientBoostingRegressor(
            n_estimators=l3_n_estimators, max_depth=l3_max_depth,
        )
        self._feature_names = None
        self._group_indices = {}

    def _resolve_indices(self, feature_names: List[str]):
        """Map feature group names to column indices."""
        self._feature_names = feature_names
        name_to_idx = {n: i for i, n in enumerate(feature_names)}
        for group_name, feat_list in HRL_FEATURE_GROUPS.items():
            indices = [name_to_idx[f] for f in feat_list if f in name_to_idx]
            self._group_indices[group_name] = indices

    def fit(self, X: np.ndarray, y: np.ndarray, feature_names: List[str]):
        self._resolve_indices(feature_names)

        # L0: M/f → Tg backbone baseline
        idx_l0 = self._group_indices.get("L0_backbone", [])
        if idx_l0:
            X_L0 = X[:, idx_l0]
            self.layer0.fit(X_L0, y)
            residual = y - self.layer0.predict(X_L0)
        else:
            residual = y.copy()

        # L1: steric correction
        idx_l1 = self._group_indices.get("L1_steric", [])
        if idx_l1:
            X_L1 = X[:, idx_l1]
            self.layer1.fit(X_L1, residual)
            residual = residual - self.layer1.predict(X_L1)

        # L2: polar correction
        idx_l2 = self._group_indices.get("L2_polar", [])
        if idx_l2:
            X_L2 = X[:, idx_l2]
            self.layer2.fit(X_L2, residual)
            residual = residual - self.layer2.predict(X_L2)

        # L3: ML residual (all features)
        self.layer3.fit(X, residual)
        return self

    def predict(self, X: np.ndarray) -> np.ndarray:
        pred = np.zeros(X.shape[0])

        idx_l0 = self._group_indices.get("L0_backbone", [])
        if idx_l0:
            pred += self.layer0.predict(X[:, idx_l0])

        idx_l1 = self._group_indices.get("L1_steric", [])
        if idx_l1:
            pred += self.layer1.predict(X[:, idx_l1])

        idx_l2 = self._group_indices.get("L2_polar", [])
        if idx_l2:
            pred += self.layer2.predict(X[:, idx_l2])

        pred += self.layer3.predict(X)
        return pred

    def diagnose(self, X: np.ndarray, y: np.ndarray) -> Dict[str, Any]:
        """Per-layer variance decomposition."""
        contributions = self.get_layer_contributions(X)
        result = {}
        cumulative = np.zeros_like(y)
        for name in ["L0", "L1", "L2", "L3"]:
            cumulative = cumulative + contributions[name]
            residual = y - cumulative
            r2 = 1 - np.sum(residual**2) / np.sum((y - y.mean())**2)
            mae = np.mean(np.abs(residual))
            result[name] = {"cumulative_R2": float(r2), "MAE": float(mae),
                            "contribution_std": float(np.std(contributions[name]))}
        return result

    def get_layer_contributions(self, X: np.ndarray) -> Dict[str, np.ndarray]:
        result = {}
        idx_l0 = self._group_indices.get("L0_backbone", [])
        result["L0"] = self.layer0.predict(X[:, idx_l0]) if idx_l0 else np.zeros(X.shape[0])
        idx_l1 = self._group_indices.get("L1_steric", [])
        result["L1"] = self.layer1.predict(X[:, idx_l1]) if idx_l1 else np.zeros(X.shape[0])
        idx_l2 = self._group_indices.get("L2_polar", [])
        result["L2"] = self.layer2.predict(X[:, idx_l2]) if idx_l2 else np.zeros(X.shape[0])
        result["L3"] = self.layer3.predict(X)
        return result


def nested_cv_hrl(
    X: np.ndarray,
    y: np.ndarray,
    feature_names: List[str],
    outer_splits: int = 5,
    outer_repeats: int = 3,
    **hrl_kwargs,
) -> Dict[str, Any]:
    """Run Nested CV with HRL model. Output format matches evaluation.py."""
    outer_cv = RepeatedKFold(n_splits=outer_splits, n_repeats=outer_repeats, random_state=42)
    fold_results = []

    for fold_idx, (train_idx, test_idx) in enumerate(outer_cv.split(X)):
        X_train, X_test = X[train_idx], X[test_idx]
        y_train, y_test = y[train_idx], y[test_idx]

        model = HierarchicalTgPredictor(**hrl_kwargs)
        model.fit(X_train, y_train, feature_names)
        y_pred = model.predict(X_test)

        r2 = 1 - np.sum((y_test - y_pred)**2) / np.sum((y_test - y_test.mean())**2)
        mae = np.mean(np.abs(y_test - y_pred))
        fold_results.append({"fold": fold_idx, "r2": float(r2), "mae": float(mae)})

    r2s = [f["r2"] for f in fold_results]
    maes = [f["mae"] for f in fold_results]
    return {
        "R2_mean": float(np.mean(r2s)),
        "R2_std": float(np.std(r2s)),
        "MAE_mean": float(np.mean(maes)),
        "MAE_std": float(np.std(maes)),
        "fold_results": fold_results,
    }
```

```python
"""
SMILES → PyG Graph with Physics-Enhanced Atom Features
物理增强分子图构建

Atom features (25 dim):
  Standard (15): element_one_hot[11], charge, degree, aromatic, in_ring
  Physics (10): is_flexible_bond, gc_group[5], is_backbone, is_sidechain,
                local_steric, is_sp3

Edge features (6 dim): bond_type[4], is_conjugated, is_in_ring
"""

import torch
from torch_geometric.data import Data
from rdkit import Chem
from src.features.virtual_polymerization import build_oligomer
from src.features.physical_proxy import _count_flexible_bonds, _find_backbone_atoms
from src.features.solubility_param import _COMPILED_PATTERNS


ELEMENTS = ["C", "N", "O", "S", "Si", "P", "F", "Cl", "Br", "I", "other"]
GC_CATEGORIES = 5  # 碳氢, 芳香, 含氧, 含氮, 卤素


def smiles_to_graph(smiles: str, n_repeat: int = 3,
                    physics_features: bool = True) -> Data:
    """Convert polymer repeat SMILES to PyG Data object."""
    # 1. 组装寡聚体
    oligo_smiles = build_oligomer(smiles, n_repeat)
    if oligo_smiles is None:
        oligo_smiles = smiles.replace("*", "[H]")

    mol = Chem.MolFromSmiles(oligo_smiles)
    if mol is None:
        raise ValueError(f"Cannot parse: {oligo_smiles}")

    # 2. 原子特征
    backbone = _find_backbone_atoms(mol, smiles) if physics_features else set()
    flexible_atoms = _get_flexible_bond_atoms(mol) if physics_features else set()

    x = []
    for atom in mol.GetAtoms():
        feat = _atom_features(atom, mol, backbone, flexible_atoms, physics_features)
        x.append(feat)
    x = torch.tensor(x, dtype=torch.float)

    # 3. 边 (双向)
    edge_index = []
    edge_attr = []
    for bond in mol.GetBonds():
        i, j = bond.GetBeginAtomIdx(), bond.GetEndAtomIdx()
        edge_index.extend([[i, j], [j, i]])
        ef = _edge_features(bond)
        edge_attr.extend([ef, ef])

    edge_index = torch.tensor(edge_index, dtype=torch.long).t().contiguous()
    edge_attr = torch.tensor(edge_attr, dtype=torch.float)

    # 4. 重复单元 mask (用于 GRIN 池化)
    ru_mask = _build_repeat_unit_mask(mol, n_repeat)

    return Data(x=x, edge_index=edge_index, edge_attr=edge_attr,
                repeat_unit_mask=torch.tensor(ru_mask, dtype=torch.long))
```

### 4.2 `src/gnn/physics_gat.py` 核心架构

```python
import torch
import torch.nn as nn
from torch_geometric.nn import GATConv, global_max_pool


class RepeatInvariantPooling(nn.Module):
    """GRIN-inspired: max-pool only middle repeat unit's atoms."""
    def forward(self, x, repeat_unit_mask, batch):
        # repeat_unit_mask: 0=left-RU, 1=middle-RU, 2=right-RU
        # 只取 middle (mask==1) 的原子做 max pool
        middle_mask = (repeat_unit_mask == 1)
        x_middle = x.clone()
        x_middle[~middle_mask] = -1e9  # mask out non-middle
        return global_max_pool(x_middle, batch)


class PhysicsGAT(nn.Module):
    """3-layer GAT with physics-enhanced features + GRIN pooling."""
    def __init__(self, in_channels=25, hidden=128, out_channels=64,
                 heads=4, dropout=0.2, edge_dim=6):
        super().__init__()
        self.conv1 = GATConv(in_channels, hidden // heads, heads=heads,
                             edge_dim=edge_dim, dropout=dropout)
        self.conv2 = GATConv(hidden, hidden // heads, heads=heads,
                             edge_dim=edge_dim, dropout=dropout)
        self.conv3 = GATConv(hidden, out_channels, heads=1,
                             edge_dim=edge_dim, dropout=dropout)
        self.bn1 = nn.BatchNorm1d(hidden)
        self.bn2 = nn.BatchNorm1d(hidden)
        self.pool = RepeatInvariantPooling()
        self.dropout = nn.Dropout(dropout)

    def forward(self, x, edge_index, edge_attr, batch, repeat_unit_mask):
        x = self.conv1(x, edge_index, edge_attr).elu()
        x = self.bn1(x)
        x = self.dropout(x)
        x = self.conv2(x, edge_index, edge_attr).elu()
        x = self.bn2(x)
        x = self.dropout(x)
        x = self.conv3(x, edge_index, edge_attr).elu()
        return self.pool(x, repeat_unit_mask, batch)  # [batch, 64]
```

### 4.3 `src/gnn/tandem_m2m.py` Tandem 架构

```python
class TandemM2M(nn.Module):
    """Tandem-M2M: Physics branch (GC) + Learning branch (GNN).

    Tg = Tg_GC + alpha * GNN_residual(graph, tabular_features)
    """
    def __init__(self, atom_dim=25, gnn_hidden=128, gnn_out=64,
                 tabular_dim=0, heads=4, dropout=0.2, edge_dim=6):
        super().__init__()
        self.gnn = PhysicsGAT(atom_dim, gnn_hidden, gnn_out, heads, dropout, edge_dim)

        mlp_in = gnn_out + tabular_dim
        self.mlp = nn.Sequential(
            nn.Linear(mlp_in, 64), nn.ELU(), nn.Dropout(dropout),
            nn.Linear(64, 32), nn.ELU(), nn.Dropout(dropout),
            nn.Linear(32, 1),
        )
        self.alpha = nn.Parameter(torch.tensor(0.5))

    def forward(self, data, tg_gc, tabular=None):
        """
        data: PyG Batch
        tg_gc: tensor [batch], Van Krevelen GC baseline
        tabular: optional tensor [batch, tabular_dim], PPF+VPD features
        """
        emb = self.gnn(data.x, data.edge_index, data.edge_attr,
                       data.batch, data.repeat_unit_mask)

        if tabular is not None:
            emb = torch.cat([emb, tabular], dim=-1)

        residual = self.mlp(emb).squeeze(-1)
        return tg_gc + self.alpha * residual

    def get_embedding(self, data):
        """Extract 64-dim GNN embedding for sklearn fusion."""
        with torch.no_grad():
            return self.gnn(data.x, data.edge_index, data.edge_attr,
                            data.batch, data.repeat_unit_mask)
```
